# Requires Android NDK toolchain
cmake_minimum_required(VERSION 3.20)

# SWIG-generated JNI wrapper + core
add_library(anychat_android SHARED
    jni/anychat_wrap.cpp    # generated by SWIG
)

target_link_libraries(anychat_android
    PRIVATE anychat_core
    PRIVATE log    # Android log library
)

target_include_directories(anychat_android
    PRIVATE ${CMAKE_SOURCE_DIR}/core/include
)
